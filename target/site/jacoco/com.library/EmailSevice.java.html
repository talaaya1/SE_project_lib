package com.library;

import jakarta.mail.*;
import jakarta.mail.internet.*;
import java.util.Properties;

/**
 * خدمة البريد الإلكتروني باستخدام Jakarta Mail وSMTP
 */
class EmailSevice {
    private final String host = "smtp.gmail.com";
    private final int port = 587;
    private final String username;  // بريدك الإلكتروني
    private final String password;  // App Password

    public EmailSevice(String username, String password) {
        this.username = username;
        this.password = password;
    }

    /**
     * إرسال بريد إلكتروني
     */
    public boolean sendEmail(String toEmail, String subject, String body) {
        // التحقق من الإعدادات
        if (username == null || password == null ||
                username.contains("your-email") || password.contains("your-password")) {
            System.out.println("❌ Please configure email settings first!");
            return false;
        }

        // إعداد خصائص SMTP
        Properties props = new Properties();
        props.put("mail.smtp.auth", "true");
        props.put("mail.smtp.starttls.enable", "true");
        props.put("mail.smtp.host", host);
        props.put("mail.smtp.port", port);
        props.put("mail.smtp.ssl.trust", host);

        // إنشاء الجلسة مع المصادقة
        Session session = Session.getInstance(props,
                new jakarta.mail.Authenticator() {
                    protected PasswordAuthentication getPasswordAuthentication() {
                        return new PasswordAuthentication(username, password);
                    }
                });

        try {
            // إنشاء الرسالة
            Message message = new MimeMessage(session);
            message.setFrom(new InternetAddress(username));
            message.setRecipients(Message.RecipientType.TO,
                    InternetAddress.parse(toEmail));
            message.setSubject(subject);

            // إضافة نص الرسالة
            message.setText(body);

            // إرسال الرسالة
            Transport.send(message);

            System.out.println("✅ Email sent successfully to: " + toEmail);
            logEmail(toEmail, subject, true, null);
            return true;

        } catch (MessagingException e) {
            System.err.println("❌ Failed to send email: " + e.getMessage());
            logEmail(toEmail, subject, false, e.getMessage());
            return false;
        }
    }

    /**
     * إرسال بريد HTML
     */
    public boolean sendHtmlEmail(String toEmail, String subject, String htmlBody) {
        Properties props = new Properties();
        props.put("mail.smtp.auth", "true");
        props.put("mail.smtp.starttls.enable", "true");
        props.put("mail.smtp.host", host);
        props.put("mail.smtp.port", port);

        Session session = Session.getInstance(props,
                new jakarta.mail.Authenticator() {
                    protected PasswordAuthentication getPasswordAuthentication() {
                        return new PasswordAuthentication(username, password);
                    }
                });

        try {
            Message message = new MimeMessage(session);
            message.setFrom(new InternetAddress(username));
            message.setRecipients(Message.RecipientType.TO,
                    InternetAddress.parse(toEmail));
            message.setSubject(subject);
            // إعداد محتوى HTML
            message.setContent(htmlBody, "text/html; charset=utf-8");

            Transport.send(message);
            System.out.println("✅ HTML email sent to: " + toEmail);
            return true;

        } catch (MessagingException e) {
            System.err.println("❌ HTML email failed: " + e.getMessage());
            return false;
        }
    }

    /**
     * تسجيل محاولات الإرسال
     */
    private void logEmail(String toEmail, String subject, boolean success, String error) {
        try {
            java.io.FileWriter fw = new java.io.FileWriter("email_log.txt", true);
            String status = success ? "SUCCESS" : "FAILED: " + error;
            fw.write(java.time.LocalDateTime.now() + " | " +
                    toEmail + " | " + subject + " | " + status + "\n");
            fw.close();
        } catch (java.io.IOException e) {
            System.err.println("Error logging email: " + e.getMessage());
        }
    }
}
