package com.library;

import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

public class User {
    private String name;
    private String email;
    private String passwordHash;
    private List<Media> borrowedMedia;
    private double totalFines;

    public User(String name, String email, String password) {
        this.name = name;
        this.email = email;
        this.passwordHash = PasswordHasher.hashPassword(password);
        this.borrowedMedia = new ArrayList<>();
        this.totalFines = 0.0;
    }

    private User(String name, String email, String passwordHash, double fines) {
        this.name = name;
        this.email = email;
        this.passwordHash = passwordHash;
        this.borrowedMedia = new ArrayList<>();
        this.totalFines = fines;
    }

    public boolean verifyPassword(String password) {
        return PasswordHasher.verifyPassword(password, this.passwordHash);
    }

    public boolean changePassword(String oldPassword, String newPassword) {
        if (verifyPassword(oldPassword)) {
            this.passwordHash = PasswordHasher.hashPassword(newPassword);
            return true;
        }
        return false;
    }

    public String getName() { return name; }
    public String getEmail() { return email; }
    public String getPasswordHash() { return passwordHash; }
    public List<Media> getBorrowedMedia() { return new ArrayList<>(borrowedMedia); }
    public double getTotalFines() { return totalFines; }

    public void setTotalFines(double fines) { this.totalFines = fines; }

    public void addFine(double amount) {
        this.totalFines += amount;
    }

    public boolean payFine(double amount) {
        if (amount > 0 && amount <= totalFines) {
            totalFines -= amount;
            return true;
        }
        return false;
    }

    public void borrowMediaRecord(Media media) {
        borrowedMedia.add(media);
    }

    public void returnMediaRecord(Media media) {
        borrowedMedia.remove(media);
    }

    public boolean canBorrow() {
        return totalFines <= 0 && borrowedMedia.size() < 5;
    }

    public String toFileString() {
        return name + "," + email + "," + passwordHash + "," + totalFines;
    }

    public static User fromFileString(String line) {
        try {
            String[] parts = line.split(",");
            if (parts.length >= 4) {
                String name = parts[0];
                String email = parts[1];
                String passwordHash = parts[2];
                double fines = Double.parseDouble(parts[3]);

                User user = new User(name, email, passwordHash, fines);
                return user;
            }
        } catch (Exception e) {
            System.err.println("Error parsing user: " + e.getMessage());
        }
        return null;
    }

    @Override
    public String toString() {
        return name + " (" + email + ") - Fines: " + totalFines + " NIS";
    }
}
