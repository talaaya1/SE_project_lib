package com.library;

import java.time.LocalDate;

public class Book extends Media {
    private String author;
    private String isbn;

    // ------------ Constructors ------------
    public Book(String title, String author, String isbn) {
        super(title);
        this.author = author;
        this.isbn = isbn;
    }

    public Book(String title, String author, String isbn, int availableCopies) {
        super(title, availableCopies); // نستخدم availableCopies مباشرة
        this.author = author;
        this.isbn = isbn;
    }

    // ------------ Getters ------------
    public String getAuthor() { return author; }
    public String getIsbn() { return isbn; }

    public String getStatus() {
        return availableCopies > 0 ? "Available" : "Not Available";
    }

    // ------------ Media overrides ------------
    @Override
    public String getType() {
        return "Book";
    }

    @Override
    public int getBorrowDays() {
        return 28;
    }

    // ===== Borrow & Return Methods =====
    @Override
    public boolean borrow(String userEmail) {
        if (availableCopies <= 0) return false;
        availableCopies--;
        borrowed = true; // أي نسخة مستعارة تجعل الكتاب "مستعار"
        borrowerEmail = userEmail;
        borrowDate = LocalDate.now();
        dueDate = LocalDate.now().plusDays(getBorrowDays());
        return true;
    }

    @Override
    public void returned() {
        availableCopies++; // زيادة النسخ عند الإرجاع
        borrowed = false;
        borrowerEmail = null;
        borrowDate = null;
        dueDate = null;
    }

    // ------------ toString ------------
    @Override
    public String toString() {
        return "Book: " + title + " by " + author + " (ISBN: " + isbn + ") - " +
                availableCopies + " copies " +
                (borrowed ? "(Borrowed by " + borrowerEmail + ", due: " + dueDate + ")" : "(Available)");
    }
}
