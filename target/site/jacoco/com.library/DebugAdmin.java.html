package com.library;

import java.io.*;
import java.util.Scanner;

public class DebugAdmin {
    private static final String ADMIN_FILE = "admins.txt";

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("=== إضافة مدير جديد ===\n");

        // 1. أخذ البيانات من المستخدم
        System.out.print("أدخل اسم المدير: ");
        String name = scanner.nextLine();

        System.out.print("أدخل الإيميل: ");
        String email = scanner.nextLine();

        System.out.print("أدخل كلمة المرور: ");
        String password = scanner.nextLine();

        // 2. تشفير كلمة المرور
        String hashedPassword = PasswordHasher.hashPassword(password);

        // 3. عرض البيانات المدخلة
        System.out.println("\n=== البيانات المدخلة ===");
        System.out.println("الاسم: " + name);
        System.out.println("الإيميل: " + email);
        System.out.println("كلمة المرور الأصلية: " + password);
        System.out.println("كلمة المرور المشفرة: " + hashedPassword);

        // 4. حفظ البيانات في الملف
        saveAdminToFile(name, email, hashedPassword);

        // 5. التأكيد
        System.out.println("\n✅ تم حفظ البيانات بنجاح في ملف: " + ADMIN_FILE);

        scanner.close();
    }

    public static void saveAdminToFile(String name, String email, String hashedPassword) {
        try (PrintWriter writer = new PrintWriter(new FileWriter(ADMIN_FILE, true))) {
            // تنسيق البيانات: اسم|إيميل|كلمة مرور مشفرة
            writer.println(name + "|" + email + "|" + hashedPassword);
            writer.flush();
            System.out.println("تم تخزين البيانات بالتنسيق: " + name + "," + email + "," + hashedPassword);
        } catch (IOException e) {
            System.err.println("❌ خطأ في حفظ البيانات: " + e.getMessage());
            e.printStackTrace();
        }
    }

    // يجب أن يكون لديك كلاس PasswordHasher
    static class PasswordHasher {
        public static String hashPassword(String password) {
            try {
                java.security.MessageDigest md = java.security.MessageDigest.getInstance("MD5");
                byte[] array = md.digest(password.getBytes());
                StringBuilder sb = new StringBuilder();
                for (byte b : array) {
                    sb.append(String.format("%02x", b));
                }
                return sb.toString();
            } catch (java.security.NoSuchAlgorithmException e) {
                return null;
            }
        }

        public static boolean verifyPassword(String inputPassword, String storedHash) {
            String hashedInput = hashPassword(inputPassword);
            return hashedInput != null && hashedInput.equals(storedHash);
        }
    }
}
