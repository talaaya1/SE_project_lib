    private void updateBorrowedFile(Media media) {
        try {
            File file = new File(BORROWED_FILE);
            List<String> lines = new ArrayList<>();

            if (file.exists()) {
                try (BufferedReader br = new BufferedReader(new FileReader(file))) {
                    String line;
                    while ((line = br.readLine()) != null) {
                        if (line.trim().isEmpty()) continue;

                        String[] parts = line.split(",");
                        if (parts.length >= 2 &&
                                parts[1].equals(media.getTitle()) &&
                                parts[0].equals(media.getType())) {
                            continue;
                        }
                        lines.add(line);
                    }
                }

                try (FileWriter fw = new FileWriter(file)) {
                    for (String line : lines) {
                        fw.write(line + "\n");
                    }
                }
            }
        } catch (IOException e) {
            System.err.println("Error updating borrowed file: " + e.getMessage());
        }
    }

    public List<Media> searchMedia(String keyword) {
        List<Media> results = new ArrayList<>();
        String lowerKeyword = keyword.toLowerCase();

        for (Media media : mediaList) {
            if (media.getTitle().toLowerCase().contains(lowerKeyword)) {
                results.add(media);
            } else if (media instanceof Book) {
                Book book = (Book) media;
                if (book.getAuthor().toLowerCase().contains(lowerKeyword) ||
                        book.getIsbn().toLowerCase().contains(lowerKeyword)) {
                    results.add(media);
                }
            } else if (media instanceof CD) {
                CD cd = (CD) media;
                if (cd.getArtist().toLowerCase().contains(lowerKeyword)) {
                    results.add(media);
                }
            }
        }

        return results;
    }
